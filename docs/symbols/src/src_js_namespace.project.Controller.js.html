  2 /**
  3  * Controller component of the Model View Controller implementation
  4  * @param {*} model The data Model for this Controller
  5  * @constructor
  6  */
  7 namespace.project.Controller = function( model ) {
  8 	
  9 	this._model = model;
 10 	
 11 	/**
 12 	 * @private
 13 	 * @type {number}
 14 	 */
 15 	this._timer;
 16 }
 17 
 18 /**
 19  * Starts the timer running
 20  */
 21 namespace.project.Controller.prototype.startTimer = function() {
 22 	
 23 	var self = this;
 24 	
 25 	var tick = function() {
 26 		self._model.setTime( new Date() );
 27 	}
 28 	
 29 	this._timer = setInterval( tick, 1000 );
 30 	tick();
 31 }
 32 
 33 /**
 34  * Stops the current time from running
 35  */
 36 namespace.project.Controller.prototype.stopTimer = function() {
 37 	clearInterval( this._timer );
 38 }
